#### 快速排序

Cpp版本：

```cpp
void quick_sort(int q[], int l, int r)
{
    if (l >= r) return;
    
    int i = l - 1, j = r + 1, x = q[l + r >> 1];
    while (i < j)
    {
        do i ++ ; while (q[i] < x);
        do j -- ; while (q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
}
```

```java
void quickSort(int[] q, int l, int r) {
    if (l >= r)	return;
    int pivot = q[(l + r) >> 1];
    int i = l - 1, j = r + 1;
    while (i < j) {
        do i++; while(q[i] < pivot);
        do j--; while(q[j] > pivot);
        if (i < j) {
            int temp = q[i];
            q[i] = q[j];
            q[j] = temp;
        }
    }
    quickSort(q, l, j);
    quickSort(q, j + 1, r);
}
```

```cpp
int a[] = {};
void merge_sort(int l, int r){
    if(l < r){
        int mid = l + r >> 1;
        merge_sort(l, mid);
        merge_sort(mid + 1, r);
        merge(l, mid, mid + 1, r);
    }
}

void merge(int ll, int lr, int rl, int rr){
    int i = ll, j = rl;
    int *temp = new int[rr - ll + 1];
    int index = 0;
    while(i <= lr && j <= rr){
        if(a[i] < a[j])  temp[index++] = a[i++];
        else  temp[index++] = a[j++];
    }
    while(i <= lr)  temp[index++] = a[i++];
    while( j <= rr) temp[index++] = a[j++];
    for(int k = 0; k < index; k ++){
        a[ll + k] = temp[k];
    }
    delete []temp;
}
```

```cpp
void merge(vector<PII> &segs)
{
    vector<PII> res;
    sort(segs.begin(), segs.end());
    int st = -2e9, ed = -2e9;
    for (auto seg : segs){

            if (st != -2e9) res.push_back({st, ed});
            st = seg.first, ed = seg.second;
```

```java
// 定义单链表类[后续优化可以添加尾节点，也可以添加泛型]
class LinkedList {
    private ListNode head; // 链表的头节点

    // 构造函数，初始化头节点为 null
    public LinkedList() {
        this.head = null;
    }

    // 在链表尾部插入一个新节点
    public void append(int val) {
        ListNode newNode = new ListNode(val);
        if (head == null) {
            head = newNode;
            return;
        }
        ListNode current = head;
        while (current.next != null) {
            current = current.next;
        }
        current.next = newNode;
    }

    // 在链表头部插入一个新节点
    public void prepend(int val) {
        ListNode newNode = new ListNode(val);
        newNode.next = head;
        head = newNode;
    }

    // 删除指定值的第一个节点
    public void delete(int val) {
        if (head == null) return;
        if (head.val == val) {
            head = head.next;
            return;
        }
        ListNode current = head;
        while (current.next != null && current.next.val != val) {
            current = current.next;
        }
        if (current.next != null) {
            current.next = current.next.next;
        }
    }

    // 查找指定值的节点
    public boolean search(int val) {
        ListNode current = head;
        while (current != null) {
            if (current.val == val) {
                return true;
            }
            current = current.next;
        }
        return false;
    }

    // 打印链表中的所有节点值
    public void printList() {
        ListNode current = head;
        while (current != null) {
            System.out.print(current.val + " -> ");
            current = current.next;
        }
        System.out.println("null");
    }
}
```

Java版本的并查集: 

```java
/*
主要操作的具体实现思路
查找（Find）：查找某个元素所在集合的代表元素，也就是找到该元素所在树的根节点。可以通过不断地沿着父节点向上查找，直到找到根节点。为了提高查找效率，可以使用路径压缩优化，即在查找过程中将节点直接连接到根节点。
合并（Union）：将两个集合合并成一个集合。通常的做法是将一个集合的根节点指向另一个集合的根节点。为了避免树的高度过高，可以使用按秩合并优化，即让秩较小的树的根节点指向秩较大的树的根节点。
*/

public class UnionFind {
    // 存储每个元素的父节点
    private int[] parent;
    // 存储每个集合的秩（树的高度）
    private int[] rank;
    // 集合的数量
    private int count;

    // 初始化并查集，每个元素自成一个集合
    public UnionFind(int n) {
        parent = new int[n];
        rank = new int[n];
        count = n;
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            rank[i] = 1;
        }
    }

    // 查找元素x所在集合的代表元素
    public int find(int x) {
        if (parent[x] != x) {
            // 路径压缩，将x直接连接到根节点
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }

    // 合并元素x和y所在的集合
    public void union(int x, int y) {
        int rootX = find(x);
        int rootY = find(y);
        if (rootX == rootY) {
            return;
        }
        if (rank[rootX] > rank[rootY]) {
            parent[rootY] = rootX;
        } else if (rank[rootX] < rank[rootY]) {
            parent[rootX] = rootY;
        } else {
            parent[rootY] = rootX;
            rank[rootX]++;
        }
        count--;
    }

    // 判断元素x和y是否属于同一个集合
    public boolean connected(int x, int y) {
        return find(x) == find(y);
    }

    // 获取集合的数量
    public int getCount() {
        return count;
    }

    public static void main(String[] args) {
        UnionFind uf = new UnionFind(5);
        uf.union(0, 1);
        uf.union(2, 3);
        System.out.println(uf.connected(0, 1)); // 输出: true
        System.out.println(uf.connected(0, 2)); // 输出: false
        System.out.println(uf.getCount());      // 输出: 3
    }
}
```

#### 拓扑排序topsort
```cpp
bool topsort()
{
    int hh = 0, tt = -1;

    // d[i] 存储点i的入度
    for (int i = 1; i <= n; i ++ )
        if (!d[i])
            q[ ++ tt] = i;
    
    while (hh <= tt)
    {
        int t = q[hh ++ ];
        for (int i = h[t]; i != -1; i = ne[i])
        {
            int j = e[i];
            if (-- d[j] == 0)
                q[ ++ tt] = j;
        }
    }

    // 如果所有点都入队了，说明存在拓扑序列；否则不存在拓扑序列。
    return tt == n - 1;
}
```

#### 堆优化版dijkstra算法
```cpp
typedef pair<int, int> PII;

int n;      // 点的数量
int h[N], w[N], e[N], ne[N], idx;       // 邻接表存储所有边
int dist[N];        // 存储所有点到1号点的距离
bool st[N];     // 存储每个点的最短距离是否已确定

// 求1号点到n号点的最短距离，如果不存在，则返回-1
int dijkstra()
{
    memset(dist, 0x3f, sizeof dist);
    dist[1] = 0;
    priority_queue<PII, vector<PII>, greater<PII>> heap;
    heap.push({0, 1});      // first存储距离，second存储节点编号

    while (heap.size())
    {
        auto t = heap.top();
        heap.pop();

        int ver = t.second, distance = t.first;

        if (st[ver]) continue;
        st[ver] = true;

        for (int i = h[ver]; i != -1; i = ne[i])
        {
            int j = e[i];
            if (dist[j] > distance + w[i])
            {
                dist[j] = distance + w[i];
                heap.push({dist[j], j});
            }
        }
    }

    if (dist[n] == 0x3f3f3f3f) return -1;
    return dist[n];
}
```

```cpp
    memset(dist, 0x3f, sizeof (dist));
    // 注意初始初始节点还是要置0
    dist[1] = 0;
    // 注意这里的循环次数是有实际物理含义的：k次表示最多只经过k条边
    // 如果题目没有限制经过边的数量，k替换为n（节点总数），不能是n-1, 可能存在负环。
    for(int i = 0; i < k; i++){
        memcpy(backup, dist, sizeof(dist));  // 备份的含义是，让此次迭代更新距离的所有结果都是来自上一次迭代。
        for(int j = 0; j < m; j ++){
            int a = edges[j].a, b = edges[j].b, w = edges[j].c;
            dist[b] = min( dist[b], backup[a] + w );		// 边权的更新方式--- 松弛操作
        }
    }
    // 这里使用> 0x3f3f3f3f /2 表示距离到不了，是因为可能存在负权回路，当无解时，可能dist[n] = 0x3f3f3f3f
    // 加上一个负权
    if(dist[n] > 0x3f3f3f3f / 2)        return 0x3f;
    return dist[n];
}
```

#### spfa判断是否存在负环 时间复杂度O(nm)
```cpp
int n;      // 总点数
int h[N], w[N], e[N], ne[N], idx;       // 邻接表存储所有边
int dist[N], cnt[N];        // dist[x]存储1号点到x的最短距离，cnt[x]存储1到x的最短路中经过的点数
bool st[N];     // 存储每个点是否在队列中

// 如果存在负环，则返回true，否则返回false。
bool spfa()
{
    // 不需要初始化dist数组
    // 原理：如果某条最短路径上有n个点（除了自己），那么加上自己之后一共有n+1个点，由抽屉原理一定有两个点相同，所以存在环。
    queue<int> q;
    for (int i = 1; i <= n; i ++ )
    {
        q.push(i);
        st[i] = true;
    }

    while (q.size())
    {
        auto t = q.front();
        q.pop();

        st[t] = false;

        for (int i = h[t]; i != -1; i = ne[i])
        {
            int j = e[i];
            if (dist[j] > dist[t] + w[i])
            {
                dist[j] = dist[t] + w[i];
                cnt[j] = cnt[t] + 1;
                if (cnt[j] >= n) return true; // 如果从1号点到x的最短路中包含至少n个点（不包括自己），则说明存在环
                if (!st[j])
                {
                    q.push(j);
                    st[j] = true;
                }
            }
        }
    }

    return false;
}
```
#### 最小生成树
##### 朴素prim版
```cpp
int n;      // n表示点数
int g[N][N];        // 邻接矩阵，存储所有边
int dist[N];        // 存储其他点到当前最小生成树的距离
bool st[N];     // 存储每个点是否已经在生成树中

// 如果图不连通，则返回INF(值是0x3f3f3f3f), 否则返回最小生成树的树边权重之和
int prim()
{
    memset(dist, 0x3f, sizeof dist);

    int res = 0;
    for (int i = 0; i < n; i ++ )
    {
        int t = -1;
        for (int j = 1; j <= n; j ++ )
            if (!st[j] && (t == -1 || dist[t] > dist[j]))
                t = j;

        if (i && dist[t] == INF) return INF;

        if (i) res += dist[t];
        st[t] = true;

        for (int j = 1; j <= n; j ++ ) dist[j] = min(dist[j], g[t][j]);
    }
    return res;
}
```

##### Kruskal算法
```cpp
int n, m;       // n是点数，m是边数
int p[N];       // 并查集的父节点数组

struct Edge     // 存储边
{
    int a, b, w;
    bool operator< (const Edge &W)const
    {
        return w < W.w;
    }
}edges[M];

int find(int x)     // 并查集核心操作
{
    if (p[x] != x) p[x] = find(p[x]);
    return p[x];
}

int kruskal()
{
    sort(edges, edges + m);

    for (int i = 1; i <= n; i ++ ) p[i] = i;    // 初始化并查集

    int res = 0, cnt = 0;
    for (int i = 0; i < m; i ++ )
    {
        int a = edges[i].a, b = edges[i].b, w = edges[i].w;

        a = find(a), b = find(b);
        if (a != b)     // 如果两个连通块不连通，则将这两个连通块合并
        {
            p[a] = b;
            res += w;
            cnt ++ ;
        }
    }

    if (cnt < n - 1) return INF;
    return res;
}
```
#### 染色法判断二分图 时间复杂度O(n + m)

```cpp
int n;      // n表示点数
```

#### 匈牙利算法求二分图最大匹配
```cpp
int n1, n2;     // n1表示第一个集合中的点数，n2表示第二个集合中的点数
int h[N], e[M], ne[M], idx;     // 邻接表存储所有边，匈牙利算法中只会用到从第一个集合指向第二个集合的边，所以这里只用存一个方向的边
int match[N];       // 存储第二个集合中的每个点当前匹配的第一个集合中的点是哪个

    {
        int j = e[i];
        if (!st[j])
        {
            st[j] = true;

            {
                match[j] = x;
                return true;
            }
        }

}
```

##### 多重背包问题

每个物品有数量限制，最多拿多少个。

状态计算：拿k个第i个物品，`f[i][j] = max(f[i-1][j - k * v[i]] + k * w[i])`。`k * v[i] <= j && k <= s[i]`

```cpp
// 朴素版本，时间复杂度约为O(n^3)
// 时间复杂度约为O(n^3)
for(int i = 1; i <= n; i ++){
    for(int j = 0; j <= m; j++){
        for(int k = 0; k <= s[i] && k *v[i] <= j; k++){				// 这一行的condition要注意
            f[i][j] = max(f[i][j], f[i-1][j - v[i] *k] + k * w[i]);
        }
    }
}

// 二进制打包优化方法：将最后一层k的循环从O(n)优化到O(logn)
// 将每种物品都打包成1, 2, 4, 8...
for(int i = 0; i < n; i ++){			// 每输入一行，就在这一行中处理f[]
    int v, w, s;
    cin >> v >> w >> s;
    for(int k = 1; k <= s; k *= 2){
        for(int j = m; j >= k * v; j --){

        }
        // 滚动数组求完，一定要记得 物品数 -= 打包里的物品个数k.
        s -= k;
    }
    if(s){

            f[j] = max(f[j], f[j - s * v] + s * w);
        }
    }
}
```
